# kev-youtube-video-clone-translate — runtime dependencies

# This file lists the Python packages needed to RUN the skill.
# It is informational; the canonical install command is:
#
#   bash kev-youtube-video-clone-translate/scripts/install-deps.sh
#
# Versions are pinned to what was tested on macOS 14 + Apple Silicon
# in June 2026. Newer minors should work; older may not.

# Core ML
torch>=2.6,<3.0           # pin major; weights_only=True default in 2.6+ needs the patch
torchaudio>=2.6,<3.0
torchcodec>=0.5           # torchaudio 2.11+ lazy-loads wavs via this

# TTS
TTS==0.22.0               # Coqui TTS — last version with xtts_v2 easy install
transformers==4.46.1      # 4.47+ removed BeamSearchScorer that TTS 0.22 imports

# Audio / voice processing
demucs>=4.0               # vocal / music separation
resemblyzer>=0.1.4        # speaker embeddings for diarization

# Implicit (TTS pulls these in, listed for clarity)
numpy>=1.26,<3.0
scipy
librosa
soundfile
